Table customers has columns: id, name, email, country

customers.id is primary key

customers.name represents name

customers.email represents email

customers.country represents country

Table products has columns: id, name, price, category_id

products.id is primary key

products.name represents name

products.price represents price

Table products joins categories ON products.category_id = categories.id

Join condition: products.category_id = categories.id

Table categories has columns: id, name

categories.id is primary key

categories.name represents name

Table orders has columns: id, customer_id, order_date, status, total_amount

orders.id is primary key

orders.status represents status

orders.total_amount represents total amount

Table orders joins customers ON orders.customer_id = customers.id

Join condition: orders.customer_id = customers.id

Table order_items has columns: id, order_id, product_id, quantity, price

order_items.id is primary key

order_items.price represents price

Table order_items joins orders ON order_items.order_id = orders.id

Join condition: order_items.order_id = orders.id

Table order_items joins products ON order_items.product_id = products.id

Join condition: order_items.product_id = products.id

To find orders from a specific country, join orders and customers using orders.customer_id = customers.id and filter customers.country

To calculate revenue, use orders.total_amount

To find customers without orders, use NOT EXISTS with orders.customer_id

orders.customer_id = customers.id is used to join orders with customers

order_items.product_id = products.id is used to join order_items with products
